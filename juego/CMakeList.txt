# Este archivo es para definir como construir, enlazar y compilar todo el proyecto.

# 1. Decimos cuál es la versión mínima de CMake (el contructor)
cmake_minimum_required(VERSION 3.10)

# 2. Damos nombre al proyecto e indicamos la versión.
project(Laberinto VERSION 0.1.0)

# 3. ¡IMPORTANTE! Aquí fijamos el estándar moderno (C++20)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 4. Le decimos a C++ dónde están los "manuales de instrucciones", el archivo .hpp
include_directories(include)

# 5. Indicamos el directorio de salida para los build y ejecutables
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_DEBUG ${CMAKE_SOURCE_DIR}/output)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY_RELEASE ${CMAKE_SOURCE_DIR}/output)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/output)

# 6. Creamos el programa final uniendo el jefe con sus ayudantes
add_excutable(LaberintoGame
    main.cpp
    src/Player.cpp
    src/MapCell.cpp
    src/GameMap.cpp
)
